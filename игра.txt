    <div id="auth-screen">

        <button onclick="saveUser()">Вход</button>

    <div id="menu-screen" style="display: none;">
        <h2>Выбери темп</h2>
        <button onclick="startGame(2)">Легко</button><br>
        <button onclick="startGame(5)">Нормально</button><br>
        <button onclick="startGame(8)">Хардкор</button>
    </div>

    <div id="ui">Счет: <span id="score">0</span></div>
    <canvas id="gameCanvas"></canvas>

<script>
    const tg = window.Telegram.WebApp;
    tg.expand();

    let canvas = document.getElementById('gameCanvas');
    let ctx = canvas.getContext('2d');
    let scoreElement = document.getElementById('score');
    let score = 0;
    let gameActive = false;
    let speed = 5;
    let player = { x: 50, y: 0, width: 30, height: 30 };
    let obstacles = [];

    function saveUser() {
        const name = document.getElementById('username').value;
        if(name.length > 1) {
            document.getElementById('auth-screen').style.display = 'none';
            document.getElementById('menu-screen').style.display = 'flex';
        }
    }

    function startGame(difficulty) {
        speed = difficulty;
        document.getElementById('menu-screen').style.display = 'none';
        document.getElementById('ui').style.display = 'block';
        canvas.style.display = 'block';
        
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        player.y = canvas.height - 100;
        player.x = canvas.width / 2 - 15;
        
        gameActive = true;
        animate();
    }

    // Управление тапом
    window.addEventListener('touchstart', (e) => {
        if(e.touches[0].clientX < canvas.width / 2) {
            player.x -= 40;
        } else {
            player.x += 40;
        }
    });

    function animate() {
        if(!gameActive) return;
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // Игрок (минималистичный квадрат)
        ctx.fillStyle = '#333';
        ctx.fillRect(player.x, player.y, player.width, player.height);

        // Генерация препятствий
        if(Math.random() < 0.02) {
            obstacles.push({ x: Math.random() * (canvas.width - 20), y: -20, w: 20, h: 20 });
        }

        // Отрисовка препятствий
        ctx.fillStyle = '#ff4757';
        obstacles.forEach((obs, index) => {
            obs.y += speed;
            ctx.fillRect(obs.x, obs.y, obs.w, obs.h);

            // Коллизия
            if (player.x < obs.x + obs.w && player.x + player.width > obs.x &&
                player.y < obs.y + obs.h && player.y + player.height > obs.y) {
                gameOver();
            }

            if(obs.y > canvas.height) {
                obstacles.splice(index, 1);
                score++;
                scoreElement.innerText = score;
            }
        });

        requestAnimationFrame(animate);
    }

    function gameOver() {
        gameActive = false;
        alert(`Финиш! Твой результат: ${score}`);
        location.reload();
    }
</script>
